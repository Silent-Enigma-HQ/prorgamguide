Проект 2: Список задач (To-Do List)


let tasks = [];

function addTask(task) {
    tasks.push(task);
    console.log("Задача добавлена:", task);
}

function removeTask(task) {
    tasks = tasks.filter(t => t !== task);
    console.log("Задача удалена:", task);
}

addTask("Почитать книгу");
removeTask("Почитать книгу");

Пример списка задач, где можно добавлять и удалять задачи.

Проект 3: Игра "Крестики-нолики"


const board = [
    [' ', ' ', ' '],
    [' ', ' ', ' '],
    [' ', ' ', ' ']
];

function makeMove(row, col, player) {
    if (board[row][col] === ' ') {
        board[row][col] = player;
        console.log(board);
    } else {
        console.log("Это место уже занято.");
    }
}

makeMove(0, 0, 'X');
makeMove(0, 1, 'O');

Пример игры "Крестики-нолики" для двух игроков.

Проект 4: Часы с обратным отсчетом


let countdown = 10;

function startCountdown() {
    const timer = setInterval(() => {
        console.log(countdown);
        countdown--;
        if (countdown < 0) {
            clearInterval(timer);
            console.log("Время вышло!");
        }
    }, 1000);
}

startCountdown();

Пример таймера с обратным отсчетом.

Проект 11: Шахматы


const board = [
    ['R', 'N', 'B', 'Q', 'K', 'B', 'N', 'R'],
    ['P', 'P', 'P', 'P', 'P', 'P', 'P', 'P'],
    [' ', ' ', ' ', ' ', ' ', ' ', ' ', ' '],
    [' ', ' ', ' ', ' ', ' ', ' ', ' ', ' '],
    [' ', ' ', ' ', ' ', ' ', ' ', ' ', ' '],
    [' ', ' ', ' ', ' ', ' ', ' ', ' ', ' '],
    ['p', 'p', 'p', 'p', 'p', 'p', 'p', 'p'],
    ['r', 'n', 'b', 'q', 'k', 'b', 'n', 'r']
];

function movePiece(from, to) {
    const [fromRow, fromCol] = from;
    const [toRow, toCol] = to;
    const piece = board[fromRow][fromCol];
    board[toRow][toCol] = piece;
    board[fromRow][fromCol] = ' ';
    console.log(board);
}

movePiece([6, 4], [4, 4]);

Пример шахматной игры с перемещением фигур.

Проект 12: Конвертер единиц


function convertLength(value, unit) {
    if (unit === 'meters') {
        return value * 100;
    } else if (unit === 'centimeters') {
        return value / 100;
    }
}

console.log(convertLength(1, 'meters'));  // 100
console.log(convertLength(100, 'centimeters'));  // 1

Пример конвертера единиц измерения.

Проект 14: Секретное сообщение


function encryptMessage(message) {
    return message.split('').reverse().join('');
}

console.log(encryptMessage("Привет мир!"));  // !рим тевирП
